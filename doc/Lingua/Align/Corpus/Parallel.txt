NAME
    Lingua::Align::Corpus::Parallel - Virtual Perl extension for reading
    parallel sentence-aligned corpora; default format is plain text (see
    Lingua::Align::Corpus::Parallel::Bitext)

SYNOPSIS
      use Lingua::Align::Corpus::Parallel;

      my $corpus = new Lingua::Align::Corpus::Parallel(-srcfile => $srcfile,
                                                       -trgfile => $trgfile);

      my @src=();
      my @trg=();
      while ($corpus->next_alignment(\@src,\@trg)){
         print "src> ";
         print join(' ',@src);
         print "\ntrg> ";
         print join(' ',@trg);
         print "============================\n";
      }
